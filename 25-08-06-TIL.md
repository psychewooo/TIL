초기 작성 코드
```python
import sys

sys.stdin = open('input.txt')

# 테스트 케이스 개수
T = int(input())

for test_case in range(1, T+1):
    N = int(input())

    # 인덱스가 있는 빈 10 * 10 격자 생성
    white_matrix = [[0] * 10 for _ in range(10)]

    # 1. 왼쪽 위와 오른쪽 아래 모서리 인덱스와 색상 정보 불러오기
    for _ in range(N):
        r1, c1, r2, c2, color = list(map(int, input().split()))

        # 2. 주어진 정보를 토대로 색칠
        # 0: 아무 것도 칠하지 않음, 1: 빨간색, 2: 파란색, 3: 보라색
        for r in range(r1, r2+1):       # r2까지이므로 range는 r2+1
            for c in range(c1, c2+1):   # c2까지이므로 range는 c2+1
                # 빨간색 색칠
                if color == 1:
                    white_matrix[r][c] += 1
                # 파란색 색칠
                elif color == 2:
                    white_matrix[r][c] += 2
        
        # 3. 보라색으로 칠한 칸 세기
        # 보라색으로 칠한 칸의 수 초기화
        purple_count = 0
        for r in range(10):
            for c in range(10):
                if white_matrix[r][c] == 3:
                    purple_count += 1

    print(f'#{test_case} {purple_count}')
  ```
